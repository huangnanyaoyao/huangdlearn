			function startTime() {
				var today = new Date();
				var h = today.getHours();
				var m = today.getMinutes();
				var s = today.getSeconds();
				m = checkTime(m);
				s = checkTime(s);
				document.getElementById("div").innerHTML = h + ":" + m + ":" + s;
				setTimeout(function() {
					startTime();
					
				}, 500);
			}
			function checkTime(i) {
				if(i < 10) {
					i = "0" + i;
				}
				return i;
			}
		</script>
